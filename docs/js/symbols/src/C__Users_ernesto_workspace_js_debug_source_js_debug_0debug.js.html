1 /**
  2  * @fileoverview This is to help proxy the different browsers console behaviors.
  3  */
  4 /*
  5 // Any debug.options defined here will be shown and rest will be hidden... unless this block isn't included, then all will be shown.
  6 var debug = {
  7 		options: {
  8 			info: true
  9 		}
 10 }*/
 11 
 12 var debug = (function(){
 13 	var _this = {};
 14 	
 15 	//set default to avoid errors
 16 	_this.info = function(){};
 17 	_this.log = function(){};
 18 	_this.debug = function(){};
 19 	_this.warn = function(){};
 20 	_this.error = function(){}; //function(msg){ alert(msg); };
 21 	_this.assert = function(){};
 22 	_this.profile = function(){};
 23 	_this.profileEnd = function(){};
 24 	_this.time = function(){};
 25 	_this.timeEnd = function(){};
 26 	
 27 	var _defaultOptions = {};
 28 	//#ifdef debug
 29 	var 
 30 		_defaultOptions = {
 31 			info: true
 32 		,	log: true
 33 		,	debug: true
 34 		,	warn: true
 35 		,	error: true
 36 		,	assert: true
 37 		,	profile: true
 38 		,	profileEnd: true
 39 		,	time: true
 40 		,	timeEnd: true
 41 		}
 42 	;
 43 	//#endif 
 44 	var 
 45 		_options =  debug.options || _defaultOptions
 46 	,	_winCon = (window.console) || false
 47 	;
 48 
 49 	// Determine Debug Mode
 50 	if(_winCon && _winCon.debug) {
 51 		/**
 52 		 * Make Debug object a reference to console if we're in
 53 		 * FireFox and the FireBug module exists or have fireBug Lite included on the page.
 54 		 */ 
 55 		_this = _winCon;
 56 		_this.log = _this.debug;
 57 		_this.options = _options;
 58 		
 59 	}	else if (_winCon && navigator.userAgent.toLowerCase().indexOf('webkit') > -1){
 60 		// Webkit Debug Console
 61 		function DebugWebKit() {}
 62 		DebugWebKit.info = function (message, objects) {
 63 			DebugWebKit.console('INFO', message, objects);
 64 		};
 65 		DebugWebKit.log = DebugWebKit.info;
 66 		DebugWebKit.assert = function (message, objects) {
 67 			DebugWebKit.console('ASSERT', Boolean(eval(message)), objects);
 68 		};
 69 		DebugWebKit.debug = function (message, objects) {
 70 			DebugWebKit.console('DEBUG', message, objects);
 71 		};
 72 		DebugWebKit.warn = function (message, objects) {
 73 				DebugWebKit.console('WARN', message, objects);
 74 		};
 75 		DebugWebKit.error = function (message, objects) {
 76 			DebugWebKit.console('ERROR', message, objects);
 77 		};
 78 		DebugWebKit.console = function(type, message, objects) {
 79 			var output = [type, ': ', message].join('');
 80 			if(objects) {
 81 				output = [output, ' Object ',  objects.toString()].join('');
 82 			}
 83 			_winCon.log(output);
 84 		};
 85 		DebugWebKit.profile = function(){};
 86 		DebugWebKit.profileEnd = function(){};
 87 		DebugWebKit.time = function(){};
 88 		DebugWebKit.timeEnd = function(){};
 89 		DebugWebKit.options = _this.options;
 90 		
 91 		_this = DebugWebKit;
 92 		
 93 	}	else if(_winCon) {
 94 		_this = _winCon;
 95 
 96 		_this.debug = function(){
 97 			var _args = [].slice.call(arguments,0);
 98 			_args.unshift("DEBUG: ");
 99 			_winCon.warn(_args.join(", "));
100 		};
101 		_this.profile = function(){};
102 		_this.profileEnd = function(){};
103 		_this.time = function(){};
104 		_this.timeEnd = function(){};
105 	
106 	}
107 	
108 	_this.options = _options;
109 
110 	/* Optional disabling of specific log types */
111 	if(!_options.info) _this.info = function(){};
112 	if(!_options.log) _this.log = function(){};
113 	if(!_options.debug) _this.debug = function(){};
114 	if(!_options.warn) _this.warn = function(){};
115 	if(!_options.error) _this.error = function(){ };
116 	if(!_options.assert) _this.assert = function(){ };
117 	if(!_options.profile) _this.profile = function(){};
118 	if(!_options.profileEnd) _this.profileEnd = function(){};
119 	if(!_options.time) _this.time = function(){};
120 	if(!_options.timeEnd) _this.timeEnd = function(){};
121 	
122 	return _this;
123 })();
